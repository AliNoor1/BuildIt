      def interp_error(interp_and_eval, x, xx, test, norm):
          """Compute norm of interpolation error for function test
          using method interp_and_eval from points x to points xx.
          """
          A = interp_and_eval(x, xx)
          y = test(x)
          yy = test(xx)
          return numpy.linalg.norm(A @ y - yy, norm)

          pyplot.xlabel('Number of points')
          pyplot.ylabel('Error')

      def vander_chebyshev(x, n=None):
          """Vandermonde matrix in Chebyshev basis."""
          if n is None:
              n = len(x)
          T = numpy.ones((len(x), n))
          if n > 1:
              T[:,1] = x
          for k in range(1,n-1):
              T[:,k+1] = 2* x * T[:,k] - T[:,k-1]
          return T

      ## Piecewise interpolation
      ​
      Instead of using global interpolation, we can perform interpolation with piecewise polynomial functions.  For example, if $\{ x_i \}_{i=0}^n$ is a strictly increasing set of points at which we have data $\{ y_i \}$, then we may find the index $j$ such that $x_{j} \le x \le x_{j+1}$ and use the interpolation
      $$ p_j(x) = y_j + \frac{y_{j+1} - y_j}{x_{j+1} - x_j} (x - x_j) . $$
      Differentiating $p(x)$ with respect to $y$ yields the evaluation matrix that we will analize.
      This is equivalent to using one of our existing `interp_and_eval` functions.

      def interp_and_eval_plinear(x, xx):
          """Construct matrix to interpolate data observed
          at points x to output points xx using piecewise
          linear interpolation.
          """
          m = len(xx)
          n = len(x)
          A = numpy.zeros((m,n))
          loc = numpy.maximum(0, x.searchsorted(xx) - 1)
          for i, j in enumerate(loc):
              # Construct the ith row of A to interpolate from
              # x[j] and x[j+1] to xx[i].
              A[i,j:j+2] = interp_and_eval_monomial(x[j:j+2], [xx[i]])
              # The following two lines produce an equivalent result
              # for this case of piecewise linear interpolation.
              # A[i,j] = 1 - (xx[i] - x[j]) / (x[j+1] - x[j])
              # A[i,j+1] = (xx[i] - x[j]) / (x[j+1] - x[j])
          return A
